int mod(int x, int y)

{
    
	int z;

    	z = x - x/y*y;
    
	return(z);

}


int gcd(int a, int b)

{
    
	if (b==0) 
		return(0);
    
	else if (mod(a,b)==0)
        
		return(b);
    
	else return(gcd(b, mod(a,b)));

}

void main(){

	int a;
	int b;
	int result;
	scanf(a);
	scanf(b);
	result = gcd(a, b);
	printf(result);
	
}
